==========
A basic footnote usage
==========
\id GEN
\c 1
\s1 The Preaching of John the Baptist
\r (Matthew 3.1-12; Luke 3.1-18; John 1.19-28)
\p
\v 1 This is the Good News about Jesus Christ, the Son of God. \f + \fr 1.1: \ft Some
manuscripts do not have \fq the Son of God.\f*
-----------
(File 
	(bookIdentification (idMarker (bookcode) (text))) 
	(chapter 
		(cMarker (chapterNumber)) 
		(sBlock (sMarker (text) (rMarker (text)))) 
		(pMarker 
			(vMarker (verseNumber)) 
			(verseText (text)) 
			(footnote 
				(fMarker (caller) (noteText (text)) (frMarker (noteText (text))) (ftMarker (noteText (text))) (fqMarker (noteText (text)))))
		)
	)
)
